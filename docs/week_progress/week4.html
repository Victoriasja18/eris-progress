from Bio import SeqIO
import pandas as pd
from pathlib import Path

def unique_is(filepath: str):
    df = pd.read_table(filepath, sep="\t")
    dim = df.shape[0]
    count = len(df.groupby("target")["target"].count())
    return f'Total IS: {dim}, Total unique IS: {count}'

def calc_seq(filepath: str):

In the pipeline.py in extract_sequence function it mentioned that they are “Extracts the full nucleotide sequence of the locus from the assembly. Handles multi-contig loci by stitching segments together in graph order”.

I want to see how does Eris create the locus and the position they retained before and after sorting

import importlib
import eris.pipeline
from eris.pipeline import Pipeline
from eris.graph import TopologyEngine 
from eris.io import GenomeAssembly, TargetDatabase
import pandas as pd
importlib.reload(eris.pipeline)

# Target database
target_db = TargetDatabase("../../data/IS_n.mmi")

# Do genome assembly
genome = GenomeAssembly.from_file("../../data/EC958_complete_genome.fasta")

# See how does the locus builder actually work
def collect_loci(genome, target_db, label):
    rows = []
    with Pipeline(target_db=target_db) as pipeline:
        for locus in pipeline(genome):
            target = locus.targets[0] if locus.targets else None
            num_segments = len(target.segments) if target else 0
            rows.append({
                "label":        label,
                "locus_id":     locus.id,
                "target":       target.id if target else None,
                "num_segments": num_segments,
                "is_stitched":  num_segments > 1,
            })
    return pd.DataFrame(rows)

sort = collect_loci(genome, target_db, label="sorted")

sort.to_csv("../result/sorted_loci.tsv", sep="\t",index= False)
